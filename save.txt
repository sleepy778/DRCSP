<button id="console-toggle-btn" 
            class="px-4 py-3 bg-gray-700 text-sm font-semibold text-white rounded-full hover:bg-gray-600 transition duration-200"
            onclick="toggleConsoleMonitor()">
        Show Console (0)
    </button>

    <!-- Main Console Monitor Container -->
    <div id="console-monitor-overlay" class="hidden h-96 w-[400px] lg:w-[600px]">
        
        <!-- Header/Controls -->
        <div class="flex justify-between items-center p-3 bg-gray-800 border-b border-gray-700">
            <h2 class="text-lg font-semibold text-white">
                <span id="log-count">0</span> Logs & Errors
            </h2>
            <div class="flex space-x-2">
                <button onclick="clearConsoleLogs()" class="text-sm text-gray-400 hover:text-white transition">
                    Clear
                </button>
                <button onclick="toggleConsoleMonitor()" class="text-gray-400 hover:text-white">
                    <!-- SVG Icon for closing -->
                    <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path></svg>
                </button>
            </div>
        </div>

        <!-- Log Display Area -->
        <div id="log-container" class="bg-gray-900 h-full overflow-y-auto p-2">
            <!-- Logs will be injected here by JavaScript -->
        </div>
    </div>

    <!-- 
        =========================================================
        JAVASCRIPT CORE LOGIC START
        =========================================================
    -->
    <script>
        // Array to store all captured logs
        const capturedLogs = [];

        // DOM elements
        const logContainer = document.getElementById('log-container');
        const monitorOverlay = document.getElementById('console-monitor-overlay');
        const toggleButton = document.getElementById('console-toggle-btn');
        const logCountSpan = document.getElementById('log-count');

        // Store original console methods
        const originalConsole = { ...console };

        /**
         * Renders the capturedLogs array to the DOM.
         */
        function renderLogs() {
            logContainer.innerHTML = ''; // Clear existing logs
            
            // Render logs in reverse chronological order (newest on top)
            capturedLogs.slice().reverse().forEach(entry => {
                const div = document.createElement('div');
                div.className = `log-entry ${entry.type}`;

                // Format timestamp
                const time = entry.timestamp.toLocaleTimeString();

                // Format message (handles object stringification)
                let message = entry.message;
                if (typeof message === 'object' && message !== null) {
                    try {
                        message = JSON.stringify(message, null, 2);
                    } catch (e) {
                        message = message.toString();
                    }
                }
                
                div.innerHTML = `<span class="text-gray-500">[${time} - ${entry.type.toUpperCase()}]</span>: ${message}`;
                logContainer.appendChild(div);
            });

            // Update the log count in the header and toggle button
            logCountSpan.textContent = capturedLogs.length;
            toggleButton.textContent = `Show Console (${capturedLogs.length})`;
            
            // Scroll to the top to see the latest log (since we are using reverse order)
            logContainer.scrollTop = 0; 
        }

        /**
         * Captures a log message and updates the state.
         * @param {string} type - The log type (log, warn, error, info).
         * @param {any} message - The log message or data.
         */
        function captureLog(type, message) {
            // Convert arguments to a single string/object for storage
            const formattedMessage = message.map(m => {
                if (typeof m === 'object' && m !== null) return JSON.stringify(m);
                return String(m);
            }).join(' ');

            capturedLogs.push({
                type: type,
                message: formattedMessage,
                timestamp: new Date()
            });
            renderLogs();
        }

        /**
         * Overrides the console methods to capture and log to the original console.
         */
        function overrideConsole() {
            ['log', 'warn', 'error', 'info'].forEach(type => {
                console[type] = function() {
                    captureLog(type, Array.from(arguments));
                    // Also call the original console function
                    originalConsole[type].apply(originalConsole, arguments);
                };
            });
        }

        /**
         * Captures global unhandled errors.
         */
        function captureGlobalErrors() {
            window.onerror = function (message, source, lineno, colno, error) {
                const errorMessage = `RUNTIME ERROR: ${message} (Source: ${source}, Line: ${lineno})`;
                // Capture as an error type
                captureLog('error', [errorMessage]);
                
                // Return false to allow the default browser error handling (optional)
                return false; 
            };
        }

        /**
         * Toggles the visibility of the console monitor overlay.
         */
        function toggleConsoleMonitor() {
            monitorOverlay.classList.toggle('hidden');
            toggleButton.classList.toggle('hidden');
        }

        /**
         * Clears all captured logs and updates the display.
         */
        function clearConsoleLogs() {
            capturedLogs.length = 0; // Clear array efficiently
            renderLogs();
        }
        
        // --- Initialization ---
        overrideConsole();
        captureGlobalErrors();
        monitorOverlay.classList.add('hidden'); // Ensure it starts hidden
    console.info("Console Monitor Initialized. Ready to capture logs...");
    
</script>
